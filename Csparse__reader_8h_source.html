271  void update_indices(size_t r, size_t first, size_t last) {
272  /* Initializing the indices upon the first request, assuming currow=0 based on initialization above.
273  * This avoids using up space for the indices if we never do row access.
274  */
275  if (indices.size() != nc) {

277  currow=0;
278  }
279 
280  /* If left/right slice are not equal to what is stored, we reset the indices,
281  * so that the code below will know to recompute them. It's too much effort
282  * to try to figure out exactly which columns need recomputing; just do them all.
283  */
284  if (first != curstart || last != curend) {
285  std::copy(p, p + nc, indices.begin());

287  }

290  * element with row number not less than 'r'. If no such element exists, it
291  * will contain the index of the first element of the next column.
292  */

294  return;
295  }

298  if (r == currow+1) {

300  for (size_t c=first; c<last; ++c, ++pIt) {
301  P& curdex = indices[c];
302  if (curdex != *pIt && static_cast<size_t>(i[curdex]) < r) {
303  ++curdex;
304  }
305  }
306  } else if (r+1 == currow) {
307  for (size_t c=first; c<last; ++c, ++pIt) {
308  P& curdex = indices[c];
309  if (curdex != *pIt && static_cast<size_t>(i[curdex-1]) >= r) {
310  --curdex;
311  }
312  }
313 
314  } else {

317  for (size_t c = first; c < last; ++c, ++pIt) {
318  indices[c] = std::lower_bound(i + indices[c], i + *pIt, r) - i;
319  }
